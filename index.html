function drawSpace() { ctx.clearRect(0, 0, canvas.width, canvas.height); // Получаем текущие значения для динамической отрисовки let v = parseInt(document.getElementById('vis').value); let c = parseInt(document.getElementById('clouds').value); // Рассчитываем коэффициент чистоты неба (0.0 - 1.0) let purity = Math.max(0, (1 - c/100) * (v/24000)); // 1. Отрисовка Туманностей (Небул) // Их яркость теперь напрямую зависит от чистоты атмосферы ctx.globalCompositeOperation = 'screen'; nebulas.forEach(n => { let g = ctx.createRadialGradient(n.x, n.y, 0, n.x, n.y, n.radius); // Умножаем прозрачность цвета на purity let colorWithAlpha = n.color.replace(/[\d\.]+\)$/g, (match) => { return (parseFloat(match) * purity).toFixed(3) + ")"; }); g.addColorStop(0, colorWithAlpha); g.addColorStop(1, 'transparent'); ctx.fillStyle = g; ctx.beginPath(); ctx.arc(n.x, n.y, n.radius, 0, Math.PI*2); ctx.fill(); }); // 2. Звездная пыль (микро-звезды) ctx.globalCompositeOperation = 'source-over'; milkyWayStars.forEach(s => { ctx.globalAlpha = s.alpha * purity; ctx.fillStyle = 'rgba(255,255,255,0.3)'; ctx.fillRect(s.x, s.y, s.size, s.size); }); // 3. Основные звезды stars.forEach(s => { s.alpha += s.scintillation * s.dir; if(s.alpha >= 1 || s.alpha <= 0.3) s.dir *= -1; // Звезды гаснут плавно в зависимости от условий ctx.globalAlpha = s.alpha * purity; ctx.fillStyle = s.color; ctx.beginPath(); ctx.arc(s.x, s.y, s.radius, 0, Math.PI*2); ctx.fill(); }); requestAnimationFrame(drawSpace); } function calculate() { let m = document.getElementById('mode').value; let c = parseInt(document.getElementById('clouds').value); let v = parseInt(document.getElementById('vis').value); let w = parseInt(document.getElementById('wind').value); document.getElementById('cloudsVal').innerText = c + '%'; document.getElementById('visVal').innerText = (v / 1000).toFixed(0) + ' км'; document.getElementById('windVal').innerText = w + ' м/с'; let score = 10; // Логика штрафов if (c > 20) score -= 2; if (c > 50) score -= 2; if (v < 8000) score -= 1; if (v < 5000) score -= 1; if (v < 2000) score -= 5; if (w > 8) score -= 1; if (w > 12) score -= 1; if (m === 'milky_way') { if (c > 30) score -= 2; if (c > 50) score -= 2; if (w > 6) score -= 1; } else if (m === 'deep_sky') { if (c > 20) score -= 3; if (c > 40) score -= 3; if (v < 10000) score -= 2; } score = Math.max(0, Math.min(10, score)); let res = document.getElementById('scoreDisplay'); let verd = document.getElementById('verdictDisplay'); res.innerText = score; // Цветовая индикация if (score >= 8) { res.style.color = '#00ffaa'; verd.innerText = "Excellent"; } else if (score >= 5) { res.style.color = '#ffcc00'; verd.innerText = "Fair"; } else { res.style.color = '#ff4444'; verd.innerText = "Poor"; } // ВАЖНО: Убираем общее управление прозрачностью канваса, // так как теперь мы управляем каждым элементом внутри drawSpace для большей мягкости. canvas.style.opacity = "1"; }
